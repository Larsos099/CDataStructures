281static inline void sl_delete_at_index(Node **rootPtrPtr, int idx) {
282 if (*rootPtrPtr == NULL || idx < 0)
283 return;
284 Node *temp = *rootPtrPtr;
285 if (idx == 0) {
286 *rootPtrPtr = temp->next;
287 free(temp->data);
288 free(temp);

292 temp = temp->next;
